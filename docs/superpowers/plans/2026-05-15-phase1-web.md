# LeaseAI Phase 1 — Web Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build polished, real-data web screens for admin (refi package trigger), financier (queue + decision), and CFO (portfolio dashboard), plus the drag-and-drop quote upload with Mistral provenance badge — all reading from and writing to the live backend.

**Architecture:** Next.js 16 App Router, server components for data fetching via `apiFetch`, server actions for mutations (POST/PATCH). No client-side data fetching for initial render. `ProvenanceBadge` shared component for "Source: Pappers" / "Extrait par Mistral" labels.

**Tech Stack:** Next.js 16 App Router · TypeScript strict · Tailwind CSS · shadcn/ui · `@supabase/ssr` · `apiFetch` from `web/lib/api-client.ts`

**Prerequisite:** Phase 1 backend plan fully deployed and accessible at `API_INTERNAL_URL`.

## Context

- Server components fetch data with `apiFetch(path, token)` — see `web/lib/api-client.ts`
- Token comes from `supabase.auth.getSession()` — see `web/app/(admin)/admin/deals/[id]/page.tsx` for pattern
- Mutations use Next.js Server Actions (`"use server"`) — no separate API routes needed
- Design tokens: navy `#0D183D`, blue `#2563EB`, teal `#10B981`, warning `#F59E0B`, danger `#EF4444`
- Typography: Satoshi for UI text, IBM Plex Mono for all numbers and financial data
- shadcn/ui components available — import from `@/components/ui/`
- Money display: always format cents → euros with `Intl.NumberFormat`

// web/lib/actions/refi-actions.ts
'use server'

import { revalidatePath } from 'next/cache'
import { createSupabaseServerClient } from '@/lib/supabase-server'
import { apiFetch } from '@/lib/api-client'
import type { RefiPackage, FinancierDecision } from '@/lib/types/refi'

async function getToken(): Promise<string> {
  const supabase = await createSupabaseServerClient()
  const { data: { session } } = await supabase.auth.getSession()
  if (!session) throw new Error('Not authenticated')
  return session.access_token
}

export async function createRefiPackage(dealId: string): Promise<{ data: RefiPackage } | { error: string }> {
  try {
    const token = await getToken()
    const result = await apiFetch<{ data: RefiPackage }>(`/deals/${dealId}/refi-packages`, token, { method: 'POST' })
    revalidatePath(`/admin/deals/${dealId}`)
    return result
  } catch (err) {
    return { error: err instanceof Error ? err.message : 'Failed to create refi package' }
  }
}

export async function sendRefiPackage(packageId: string, dealId: string): Promise<{ data: RefiPackage } | { error: string }> {
  try {
    const token = await getToken()
    const result = await apiFetch<{ data: RefiPackage }>(`/refi-packages/${packageId}/send`, token, { method: 'POST' })
    revalidatePath(`/admin/deals/${dealId}`)
    return result
  } catch (err) {
    return { error: err instanceof Error ? err.message : 'Failed to send package' }
  }
}

export async function recordFinancierDecision(
  packageId: string,
  decision: 'approved' | 'rejected',
  reason?: string
): Promise<{ data: FinancierDecision } | { error: string }> {
  try {
    const token = await getToken()
    const result = await apiFetch<{ data: FinancierDecision }>(
      `/refi-packages/${packageId}/decision`,
      token,
      { method: 'POST', body: JSON.stringify({ decision, reason }) }
    )
    revalidatePath('/financier/packages')
    return result
  } catch (err) {
    return { error: err instanceof Error ? err.message : 'Failed to record decision' }
  }
}
```

// web/components/admin/RefiPackagePanel.tsx
'use client'

import { useState, useTransition } from 'react'
import { createRefiPackage, sendRefiPackage } from '@/lib/actions/refi-actions'
import { MoneyAmount } from '@/components/shared/MoneyAmount'
import type { Deal } from '@/lib/types/admin'
import type { RefiPackage } from '@/lib/types/refi'

interface Props {
  deal: Deal
  existingPackage: RefiPackage | null
}

const STATUS_LABELS: Record<string, { label: string; color: string }> = {
  draft: { label: 'Brouillon', color: 'bg-gray-100 text-gray-600' },
  sent: { label: 'Envoyé au financier', color: 'bg-blue-50 text-blue-700' },
  approved: { label: 'Approuvé', color: 'bg-teal-50 text-teal-700' },
  rejected: { label: 'Refusé', color: 'bg-red-50 text-red-700' },
}

export function RefiPackagePanel({ deal, existingPackage }: Props) {
  const [pkg, setPkg] = useState<RefiPackage | null>(existingPackage)
  const [error, setError] = useState<string | null>(null)
  const [isPending, startTransition] = useTransition()

  const canCreate = deal.status === 'pre_approved' && pkg === null
  const canSend = pkg?.status === 'draft'

  function handleCreate() {
    setError(null)
    startTransition(async () => {
      const result = await createRefiPackage(deal.id)
      if ('error' in result) {
        setError(result.error)
      } else {
        setPkg(result.data)
      }
    })
  }

  function handleSend() {
    if (!pkg) return
    setError(null)
    startTransition(async () => {
      const result = await sendRefiPackage(pkg.id, deal.id)
      if ('error' in result) {
        setError(result.error)
      } else {
        setPkg(result.data)
      }
    })
  }

  return (
    <div className="bg-white rounded-xl border border-gray-200 p-6 mb-6">
      <h2 className="text-base font-semibold text-navy-900 mb-4">Package de refinancement</h2>

      {pkg ? (
        <div className="space-y-3">
          <div className="flex items-center justify-between">
            <span className="text-sm text-gray-500">Statut</span>
            <span className={`text-xs font-medium px-2.5 py-1 rounded-full ${STATUS_LABELS[pkg.status]?.color ?? 'bg-gray-100 text-gray-600'}`}>
              {STATUS_LABELS[pkg.status]?.label ?? pkg.status}
            </span>
          </div>
          <div className="flex items-center justify-between">
            <span className="text-sm text-gray-500">Montant</span>
            <MoneyAmount cents={pkg.amount_cents} className="text-sm" />
          </div>
          <div className="flex items-center justify-between">
            <span className="text-sm text-gray-500">Durée</span>
            <span className="text-sm font-mono">{pkg.duration_months} mois</span>
          </div>
          <div className="flex items-center justify-between">
            <span className="text-sm text-gray-500">Loyer mensuel</span>
            <MoneyAmount cents={pkg.monthly_payment_cents} className="text-sm" />
          </div>
          <div className="flex items-center justify-between">
            <span className="text-sm text-gray-500">Score risque</span>
            <span className="text-sm font-mono">{pkg.risk_band ?? '—'} ({pkg.risk_score?.toFixed(1) ?? '—'})</span>
          </div>

          {canSend && (
            <button
              onClick={handleSend}
              disabled={isPending}
              className="mt-4 w-full rounded-lg bg-blue-600 px-4 py-2 text-sm font-medium text-white hover:bg-blue-700 disabled:opacity-50 transition-colors"
            >
              {isPending ? 'Envoi…' : 'Envoyer au financier'}
            </button>
          )}
        </div>
      ) : (
        <div className="text-center py-6">
          <p className="text-sm text-gray-400 mb-4">Aucun package créé</p>
          {canCreate && (
            <button
              onClick={handleCreate}
              disabled={isPending}
              className="rounded-lg bg-navy-900 px-5 py-2 text-sm font-medium text-white hover:bg-navy-800 disabled:opacity-50 transition-colors"
              style={{ backgroundColor: '#0D183D' }}
            >
              {isPending ? 'Création…' : 'Générer le package refi'}
            </button>
          )}
        </div>
      )}

      {error && (
        <p className="mt-3 text-xs text-red-600 bg-red-50 rounded-lg px-3 py-2">{error}</p>
      )}
    </div>
  )
}
```

// web/components/admin/QuoteUploadZone.tsx
'use client'

import { useRef, useState } from 'react'
import { ProvenanceBadge } from '@/components/shared/ProvenanceBadge'
import { MoneyAmount } from '@/components/shared/MoneyAmount'
import type { Quote } from '@/lib/types/admin'

interface Props {
  dealId: string
  token: string
  onExtracted?: (quote: Quote) => void
}

const API_BASE = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:8000'

export function QuoteUploadZone({ dealId, token, onExtracted }: Props) {
  const inputRef = useRef<HTMLInputElement>(null)
  const [isDragging, setIsDragging] = useState(false)
  const [isExtracting, setIsExtracting] = useState(false)
  const [extracted, setExtracted] = useState<Quote | null>(null)
  const [error, setError] = useState<string | null>(null)

  async function handleFile(file: File) {
    if (!file.type.includes('pdf')) {
      setError('Seuls les fichiers PDF sont acceptés')
      return
    }
    setError(null)
    setIsExtracting(true)
    try {
      const formData = new FormData()
      formData.append('file', file)
      const resp = await fetch(`${API_BASE}/deals/${dealId}/quotes/upload-and-extract`, {
        method: 'POST',
        headers: { Authorization: `Bearer ${token}` },
        body: formData,
      })
      if (!resp.ok) {
        const body = await resp.json().catch(() => ({}))
        throw new Error(body?.error?.message ?? `HTTP ${resp.status}`)
      }
      const { data } = await resp.json()
      setExtracted(data)
      onExtracted?.(data)
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Échec de l\'extraction')
    } finally {
      setIsExtracting(false)
    }
  }

  function onDrop(e: React.DragEvent) {
    e.preventDefault()
    setIsDragging(false)
    const file = e.dataTransfer.files[0]
    if (file) handleFile(file)
  }

  function onInputChange(e: React.ChangeEvent<HTMLInputElement>) {
    const file = e.target.files?.[0]
    if (file) handleFile(file)
  }

  if (extracted) {
    const items = extracted.extraction_payload?.items ?? []
    return (
      <div className="border border-teal-200 rounded-xl p-5 bg-teal-50">
        <div className="flex items-center justify-between mb-4">
          <h3 className="text-sm font-semibold text-teal-900">Devis extrait</h3>
          <ProvenanceBadge source={extracted.extraction_source} />
        </div>
        <p className="text-sm font-medium text-gray-800 mb-1">{extracted.supplier_name ?? '—'}</p>
        <p className="text-xs text-gray-400 mb-4 font-mono">{extracted.quote_number ?? ''}</p>
        <div className="space-y-2 mb-4">
          {items.slice(0, 5).map((item, i) => (
            <div key={i} className="flex items-center justify-between text-xs">
              <span className="text-gray-700 truncate max-w-[55%]">{item.label}</span>
              <span className="text-gray-400 font-mono">×{item.quantity}</span>
              <MoneyAmount cents={item.total_price_cents} className="text-xs text-gray-700" />
            </div>
          ))}
          {items.length > 5 && (
            <p className="text-xs text-gray-400">+{items.length - 5} articles</p>
          )}
        </div>
        <div className="flex items-center justify-between pt-3 border-t border-teal-200">
          <span className="text-xs text-gray-500">Total HT</span>
          <MoneyAmount cents={extracted.amount_excl_tax_cents} className="text-sm font-semibold text-teal-800" />
        </div>
      </div>
    )
  }

  return (
    <div
      onDragOver={(e) => { e.preventDefault(); setIsDragging(true) }}
      onDragLeave={() => setIsDragging(false)}
      onDrop={onDrop}
      onClick={() => inputRef.current?.click()}
      className={`cursor-pointer rounded-xl border-2 border-dashed p-8 text-center transition-colors ${
        isDragging
          ? 'border-blue-400 bg-blue-50'
          : 'border-gray-200 hover:border-gray-300 hover:bg-gray-50'
      }`}
    >
      <input ref={inputRef} type="file" accept=".pdf" className="hidden" onChange={onInputChange} />
      {isExtracting ? (
        <div className="space-y-2">
          <div className="h-5 w-5 mx-auto animate-spin rounded-full border-2 border-blue-600 border-t-transparent" />
          <p className="text-sm text-blue-600 font-medium">Extraction Mistral en cours…</p>
        </div>
      ) : (
        <div className="space-y-2">
          <svg className="mx-auto h-8 w-8 text-gray-300" fill="none" viewBox="0 0 24 24" stroke="currentColor">
            <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={1.5} d="M9 13h6m-3-3v6m5 5H7a2 2 0 01-2-2V5a2 2 0 012-2h5.586a1 1 0 01.707.293l5.414 5.414a1 1 0 01.293.707V19a2 2 0 01-2 2z" />
          </svg>
          <p className="text-sm font-medium text-gray-600">Déposer le devis PDF</p>

// web/components/financier/DecisionButtons.tsx
'use client'

import { useState, useTransition } from 'react'
import { useRouter } from 'next/navigation'
import { recordFinancierDecision } from '@/lib/actions/refi-actions'

interface Props { packageId: string }

export function DecisionButtons({ packageId }: Props) {
  const router = useRouter()
  const [isPending, startTransition] = useTransition()
  const [showReject, setShowReject] = useState(false)
  const [reason, setReason] = useState('')
  const [error, setError] = useState<string | null>(null)

  function handleApprove() {
    setError(null)
    startTransition(async () => {
      const result = await recordFinancierDecision(packageId, 'approved')
      if ('error' in result) {
        setError(result.error)
      } else {
        router.push('/financier/packages')
        router.refresh()
      }
    })
  }

  function handleReject() {
    if (!reason.trim()) { setError('Le motif est obligatoire pour un refus'); return }
    setError(null)
    startTransition(async () => {
      const result = await recordFinancierDecision(packageId, 'rejected', reason.trim())
      if ('error' in result) {
        setError(result.error)
      } else {
        router.push('/financier/packages')
        router.refresh()
      }
    })
  }

  return (
    <div className="bg-white rounded-xl border border-gray-200 p-6 space-y-4">
      <h2 className="text-base font-semibold text-gray-900">Décision</h2>

      {!showReject ? (
        <div className="flex gap-3">
          <button
            onClick={handleApprove}
            disabled={isPending}
            className="flex-1 rounded-lg bg-teal-600 px-4 py-3 text-sm font-semibold text-white hover:bg-teal-700 disabled:opacity-50 transition-colors"
          >
            {isPending ? 'En cours…' : '✓ Approuver'}
          </button>
          <button
            onClick={() => setShowReject(true)}
            disabled={isPending}
            className="flex-1 rounded-lg border border-red-200 px-4 py-3 text-sm font-semibold text-red-600 hover:bg-red-50 disabled:opacity-50 transition-colors"
          >
            ✕ Refuser
          </button>
        </div>
      ) : (
        <div className="space-y-3">
          <textarea
            value={reason}
            onChange={e => setReason(e.target.value)}
            placeholder="Motif du refus (obligatoire)…"
            rows={3}
            className="w-full rounded-lg border border-gray-200 px-3 py-2 text-sm text-gray-800 placeholder:text-gray-400 focus:border-red-300 focus:outline-none resize-none"
          />
          <div className="flex gap-3">
            <button
              onClick={handleReject}
              disabled={isPending}
              className="flex-1 rounded-lg bg-red-600 px-4 py-2 text-sm font-semibold text-white hover:bg-red-700 disabled:opacity-50 transition-colors"
            >
              {isPending ? 'En cours…' : 'Confirmer le refus'}
            </button>
            <button
              onClick={() => { setShowReject(false); setReason('') }}
              className="rounded-lg border border-gray-200 px-4 py-2 text-sm text-gray-500 hover:bg-gray-50 transition-colors"
            >
              Annuler
            </button>
          </div>
        </div>
      )}

      {error && <p className="text-xs text-red-600 bg-red-50 rounded-lg px-3 py-2">{error}</p>}
    </div>
  )
}
